生成Marker的Matleb代码

N=600;
M=800;
pattern=0;
count=0;
m=4;
while m<M
    n=round(N/M*m);
    noise=rand(n,m);
    noise=imresize(noise,[N,M]);
    pattern=pattern+noise;
    count=count+1;
    m=m*2;
end
pattern=pattern./count;
pattern=histeq(pattern);
imshow(pattern);
